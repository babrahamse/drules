rule "harvest genre terms"
when
     (TRUE)
then
     copyField "655.a" to "901.a" if(exists "655.{*,0}")
     copyField "655.a" to "901.a" if(exists "655.{*,7}")
     
    replaceContents "901.a.\\." with " "
    replaceContents "901.a.," with " "
    replaceContents "901.a.(" with " "
    replaceContents "901.a.)" with " "
    replaceContents "901.a.:" with " "
    replaceContents "901.a.;" with " "
    replaceContents "901.a.  " with " "
    replaceContents "901.a.  " with " "

     copyField "901.a" to "965.a"
     removeField "901"
     combineFields "965" excluding ""
end
